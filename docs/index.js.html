 * Función encagarda de hacer la validación del formulario al cambiar el foco de un input
 * @param {Event} e - Evento que causa la llamada
 */
function check(e){
    e.target.setCustomValidity('')
    if(e.target.closest('p').getElementsByTagName('div')[0]){
        e.target.closest('p').getElementsByTagName('div')[0].remove()
    }
    let div=document.createElement('div')
    e.target.closest('p').append(div)
    if(!e.target.checkValidity()){
        
        if(e.target.validity.valueMissing){
            e.target.setCustomValidity('Campo obligatorio')
        }
        if(e.target.validity.typeMismatch){
            e.target.setCustomValidity('Introduce el tipo de dato requerido')
        }
        if(e.target.validity.patternMismatch){
            e.target.setCustomValidity('El valor introducido no cumple con lo pedidoo')
        }
        if(e.target.validity.tooShort ){
            e.target.setCustomValidity('Debes introducir almenos '+e.target.minLength+" caracteres")
        }
        if(e.target.validity.tooLong ){
            e.target.setCustomValidity('Debes introducir como maximo '+e.target.maxLength+" caracteres")
        }
        if(e.target.validity.rangeUnderflow  ){
            e.target.setCustomValidity('El valor minimo es '+e.target.min)
        }
        if(e.target.validity.rangeOverflow  ){
            e.target.setCustomValidity('El valor maximo es '+e.target.max)
        }
        div.textContent=e.target.validationMessage
        div.classList.add('red')
    }
}

/**
 * Función que asigna los listener correspondientes a las nuevas filas
 * @param  tr - Fila a la que se le añade los listeners
 */
function asignaListener(tr){
    tr.children[5].children[3].addEventListener('click',clickDel)
    tr.children[5].children[0].addEventListener('click',clickAumentar)
    tr.children[5].children[1].addEventListener('click',clickDisminuir)

    vaciaFormulario()
}
/**
 * Función que inicia el flujo para la creación de un nuevo producto
 * @param {Event} e - Evento que inicia la función 
 */
function clickAdd(e){
    e.preventDefault()
    let inputid=document.getElementById('newprod-id')
    let id=inputid.value
    let inputName=document.getElementById('newprod-name')
    let name=inputName.value
    let inputPrice=(document.getElementById('newprod-price'))
    let price=Number(inputPrice.value)
    let inputUnits=document.getElementById('newprod-units')
    let units=inputUnits.value
    if(inputPrice.checkValidity()&&inputid.checkValidity()&&inputUnits.checkValidity()&&inputName.checkValidity()){
        if(stroreApp.findProduct(id)){
            stroreApp.throwErrorPersonalizado('No puede haber dos ids iguales')
        }else{
            let tr=stroreApp.addProductToApi(id,name,price,units)
            asignaListener(tr)
        }
        
    }else{
        let invalidos=[]
        let inputs=document.getElementById('newprod').getElementsByTagName('input')
        for (const input of inputs) {
            if(!input.checkValidity()){
                invalidos.push(input)
                submitCheck(input)
            }
        }
        console.log(invalidos[0])
        invalidos[0].focus()
    }
    
}

function clickEdit(){
    document.getElementById('newprod-add').hidden=true
    let botonEdit=document.getElementById('editprod')
    botonEdit.hidden=false
    let inputId=this.closest('tr').children[0]
    let id=(inputId.textContent)

    let units=Number(this.closest('tr').children[2].textContent)

    document.getElementById('newprod-price').value=price
    document.getElementById('newprod-units').value=units
    botonEdit.addEventListener('click',edit)

    botonReset.addEventListener('click',()=>{
        document.getElementById('newprod-id').value=id
        document.getElementById('newprod-name').value=name
        document.getElementById('newprod-price').value=price
        document.getElementById('newprod-units').value=units
        reset()
        
    })
    
    
}

/**
 * Función que inicia el flujo de edición de producto
 * @param {Event} e - Evento que inicia la función
 */
function edit(e){
    e.preventDefault()
    let inputs=document.getElementsByTagName('input')
    let inputId=document.getElementById('newprod-id')
    let id= (inputId.value)
    let name=document.getElementById('newprod-name').value
    let price=Number(document.getElementById('newprod-price').value).toFixed(2)
    let units=Number(document.getElementById('newprod-units').value)
    try{
        let invalidos=[];
        for(const input of inputs){
            if(!input.checkValidity()){
                invalidos.push(input);
                submitCheck(input);
            }
        }
        if(invalidos.length> 0){
            stroreApp.throwErrorPersonalizado('El producto no ha sido editado, verifique los datos introducidos');
            invalidos[0].focus();
            return; 
        }
        stroreApp.editApi(id,units,name,price);
        document.getElementById('newprod-id').disabled=false;
    }catch(error){
        console.error(error)
    }
    document.getElementById('newprod-add').hidden=false
    document.getElementById('editprod').hidden=true
    vaciaFormulario()
}
/**
 * Función que vacia por completo los valores del formulario

    document.getElementById('newprod-id').value=''
    document.getElementById('newprod-name').value=''
    document.getElementById('newprod-price').value=''
    document.getElementById('newprod-units').value=''
}
